Aadhaar verify — services/api/src/routes/aadhaar.ts
import { Router } from "express";
import { z } from "zod";
import aadhaarData from "../data/aadhaar.json" assert { type: "json" };

const router = Router();

const Req = z.object({
  aadhaarId: z.string().min(4),
  name: z.string().min(2),
  dob: z.string().regex(/^\d{4}-\d{2}-\d{2}$/)
});

router.post("/verify", (req, res) => {
  const parsed = Req.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

  const { aadhaarId, name, dob } = parsed.data;
  const record = (aadhaarData as any[]).find((r) => r.aadhaarId === aadhaarId);

  if (!record) return res.json({ verified: false, reason: "NOT_FOUND" });

  if (record.name.toLowerCase() !== name.toLowerCase())
    return res.json({ verified: false, reason: "NAME_MISMATCH" });

  if (record.dob !== dob) return res.json({ verified: false, reason: "DOB_MISMATCH" });

  return res.json({ verified: true, reason: "MATCH" });
});

export default router;

Appointments — services/api/src/routes/appointments.ts
import { Router } from "express";
import { z } from "zod";
import { prisma } from "../db/prisma";
import centers from "../data/centers.json" assert { type: "json" };

const router = Router();

const CreateReq = z.object({
  aadhaarId: z.string(),
  patientName: z.string(),
  phone: z.string(),
  language: z.string(),
  centerId: z.string(),
  slotTime: z.string(),
  vaccineType: z.string(),
  aadhaarVerified: z.boolean()
});

function makeAppointmentId() {
  return APT-${Math.floor(100000 + Math.random() * 900000)};
}

router.post("/", async (req, res) => {
  const parsed = CreateReq.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

  const c = (centers as any[]).find((x) => x.centerId === parsed.data.centerId);
  if (!c) return res.status(404).json({ error: "Center not found" });

  const appointmentId = makeAppointmentId();
  const appt = await prisma.appointment.create({
    data: {
      appointmentId,
      aadhaarId: parsed.data.aadhaarId,
      patientName: parsed.data.patientName,
      phone: parsed.data.phone,
      language: parsed.data.language,
      centerId: parsed.data.centerId,
      centerName: c.name,
      slotTime: new Date(parsed.data.slotTime),
      vaccineType: parsed.data.vaccineType,
      aadhaarVerified: parsed.data.aadhaarVerified,
      status: "BOOKED"
    }
  });

  res.json({ appointment: appt });
});

router.get("/", async (req, res) => {
  const centerId = req.query.centerId?.toString();
  const date = req.query.date?.toString(); // YYYY-MM-DD

  const where: any = {};
  if (centerId) where.centerId = centerId;

  if (date) {
    const start = new Date(${date}T00:00:00);
    const end = new Date(${date}T23:59:59);
    where.slotTime = { gte: start, lte: end };
  }

  const list = await prisma.appointment.findMany({
    where,
    orderBy: { slotTime: "asc" }
  });

  res.json({ appointments: list });
});

router.patch("/:appointmentId", async (req, res) => {
  const StatusReq = z.object({ status: z.string() });
  const parsed = StatusReq.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

  const updated = await prisma.appointment.update({
    where: { appointmentId: req.params.appointmentId },
    data: { status: parsed.data.status }
  });

  res.json({ appointment: updated });
});

export default router;

  if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

  const shipmentId = SHIP-${Math.floor(10000 + Math.random() * 90000)};
  const batchNo = BATCH-${new Date().getFullYear()}-${Math.floor(100 + Math.random() * 900)};

  await prisma.shipment.create({
    data: {
      shipmentId,
      manufacturer: parsed.data.manufacturer,
      batchNo,
      vaccineType: parsed.data.vaccineType,
      quantity: parsed.data.quantity,
      originName: "Central Warehouse",
      originLat: 19.0760,
      originLng: 72.8777,
      destinationCenterId: parsed.data.centerId,
      status: "CREATED",
      currentLat: 19.0760,
      currentLng: 72.8777,
      currentCheckpoint: "Central Warehouse"
    }
  });

  // Basic checkpoints
  const cps = [
    { name: "Central Warehouse", lat: 19.0760, lng: 72.8777, etaSec: 0 },
    { name: "Hub A", lat: 19.10, lng: 72.86, etaSec: 10 },
    { name: "Hub B", lat: 19.115, lng: 72.852, etaSec: 20 },
    { name: "Clinic", lat: 19.1197, lng: 72.8464, etaSec: 30 }
  ];

  for (const cp of cps) {
    await prisma.shipmentCheckpoint.create({ data: { shipmentId, ...cp } });
  }

  res.json({ shipmentId, status: "CREATED" });
});

Simulation timer — services/api/src/routes/sim.ts
import { Router } from "express";
import { prisma } from "../db/prisma";

const router = Router();
const timers = new Map<string, NodeJS.Timeout>();

router.post("/start", async (req, res) => {
  const shipmentId = req.query.shipmentId?.toString();
  if (!shipmentId) return res.status(400).json({ error: "shipmentId required" });

  const ship = await prisma.shipment.findUnique({ where: { shipmentId } });
  if (!ship) return res.status(404).json({ error: "Shipment not found" });

  if (timers.has(shipmentId)) return res.json({ ok: true, message: "Already running" });

  const cps = await prisma.shipmentCheckpoint.findMany({ where: { shipmentId }, orderBy: { etaSec: "asc" } });
  let idx = 0;

  await prisma.shipment.update({ where: { shipmentId }, data: { status: "IN_TRANSIT" } });

  const t = setInterval(async () => {
    idx++;
    if (idx >= cps.length) {
      clearInterval(t);
      timers.delete(shipmentId);
      await prisma.shipment.update({
        where: { shipmentId },
        data: {
          status: "ARRIVED",
          currentCheckpoint: cps[cps.length - 1].name,
          currentLat: cps[cps.length - 1].lat,
          currentLng: cps[cps.length - 1].lng,
          lastUpdateTs: new Date()
        }
      });
      return;
    }

    const cp = cps[idx];
    await prisma.shipment.update({
      where: { shipmentId },
      data: {
        currentCheckpoint: cp.name,
        currentLat: cp.lat,
        currentLng: cp.lng,
        lastUpdateTs: new Date()
      }
    });
  }, 2000);

  timers.set(shipmentId, t);
  res.json({ ok: true, message: "Simulation started" });
});

router.post("/stop", async (req, res) => {
  const shipmentId = req.query.shipmentId?.toString();
  if (!shipmentId) return res.status(400).json({ error: "shipmentId required" });

  const t = timers.get(shipmentId);
  if (t) { clearInterval(t); timers.delete(shipmentId); }
  res.json({ ok: true });
});

export default router;

Chatbot wrapper (intent engine) — services/api/src/routes/ai.ts
import { Router } from "express";
import { z } from "zod";

const router = Router();

const ChatReq = z.object({
  message: z.string().min(1),
  language: z.string().optional()
});

router.post("/chat", async (req, res) => {
  const parsed = ChatReq.safeParse(req.body);
  if (!parsed.success) return res.status(400).json({ error: parsed.error.flatten() });

  const msg = parsed.data.message.toLowerCase();

  // Minimal intent handling (hackathon-safe)
  if (msg.includes("center") || msg.includes("near")) {
    return res.json({ reply: "I can help you find the nearest vaccination center. Open Centers screen." });
  }
  if (msg.includes("book") || msg.includes("appointment")) {
    return res.json({ reply: "Sure. Please verify Aadhaar and pick a center + time slot to book." });
  }
  if (msg.includes("document") || msg.includes("aadhaar")) {
    return res.json({ reply: "You need Aadhaar details for verification (demo uses simulated database)." });
  }

  return res.json({ reply: "I can help with centers, booking appointments, and vaccine information." });
});
